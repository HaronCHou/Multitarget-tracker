99 std::vector<bbox_t> detect_resized(image_t img, int init_w, int init_h, float thresh = 0.2, bool use_mean = false)
100 {
101 if (img.data == NULL)
102 throw std::runtime_error("Image is empty");
103 auto detection_boxes = detect(img, thresh, use_mean);
104 float wk = (float)init_w / img.w, hk = (float)init_h / img.h;
105 for (auto &i : detection_boxes) i.x *= wk, i.w *= wk, i.y *= hk, i.h *= hk;
106 return detection_boxes;
107 }

146 static image_t mat_to_image_custom(cv::Mat mat)
147 {
148 int w = mat.cols;
149 int h = mat.rows;
150 int c = mat.channels();
151 image_t im = make_image_custom(w, h, c);
152 unsigned char *data = (unsigned char *)mat.data;
153 int step = mat.step;
154 for (int y = 0; y < h; ++y) {
155 for (int k = 0; k < c; ++k) {
156 for (int x = 0; x < w; ++x) {
157 im.data[k*w*h + y*w + x] = data[y*step + x*c + k] / 255.0f;
158 }
159 }
160 }
161 return im;
162 }

345 if (sync_PyrLKOpticalFlow_gpu.empty()) {
346 std::cout << "sync_PyrLKOpticalFlow_gpu isn't initialized \n";

354 if (dst_mat_gpu.cols == 0) {
355 dst_mat_gpu = cv::cuda::GpuMat(dst_mat.size(), dst_mat.type());

366 cv::cuda::setDevice(old_gpu_id);
367 return cur_bbox_vec;

371 sync_PyrLKOpticalFlow_gpu->calc(src_grey_gpu, dst_grey_gpu, prev_pts_flow_gpu, cur_pts_flow_gpu, status_gpu, err_gpu, stream); // OpenCV 3.x

400 cur_bbox_vec[i].x += moved_x + 0.5;
401 cur_bbox_vec[i].y += moved_y + 0.5;
402 result_bbox_vec.push_back(cur_bbox_vec[i]);
403 }

413

415 cv::cuda::setDevice(old_gpu_id);

732 kf.transitionMatrix.at<float>(2) = dT;
733 kf.transitionMatrix.at<float>(9) = dT;

735
736 void set(bbox_t box)
737 {
738 initialize_kalman();
739
740 kf.errorCovPre.at<float>(0) = 1; // px
741 kf.errorCovPre.at<float>(7) = 1; // px
742 kf.errorCovPre.at<float>(14) = 1;
743 kf.errorCovPre.at<float>(21) = 1;
744 kf.errorCovPre.at<float>(28) = 1; // px
745 kf.errorCovPre.at<float>(35) = 1; // px
746
747 state.at<float>(0) = box.x;
748 state.at<float>(1) = box.y;
749 state.at<float>(2) = 0;
750 state.at<float>(3) = 0;
751 state.at<float>(4) = box.w;
752 state.at<float>(5) = box.h;
753 // <<<< Initialization
754
755 kf.statePost = state;
756 }
757
758 // Kalman.correct() calculates: statePost = statePre + gain * (z(k)-measurementMatrix*statePre);
759 // corrected state (x(k)): x(k)=x'(k)+K(k)*(z(k)-H*x'(k))
760 void correct(bbox_t box) {
761 meas.at<float>(0) = box.x;
762 meas.at<float>(1) = box.y;
763 meas.at<float>(2) = box.w;
764 meas.at<float>(3) = box.h;
765
766 kf.correct(meas);
767
768 bbox_t new_box = predict();
769 if (new_box.w == 0 || new_box.h == 0) {
770 set(box);
771 //std::cerr << " force set(): track_id = " << box.track_id <<
772 // ", x = " << box.x << ", y = " << box.y << ", w = " << box.w << ", h = " << box.h << std::endl;
773 }
774 }

776 // Kalman.predict() calculates: statePre = TransitionMatrix * statePost;
777 // predicted state (x'(k)): x(k)=A*x(k-1)+B*u(k)

779 bbox_t box;
780 state = kf.predict();
781
782 box.x = state.at<float>(0);
783 box.y = state.at<float>(1);
784 box.w = state.at<float>(4);
785 box.h = state.at<float>(5);
786 return box;
787 }
788
789 void initialize_kalman()
790 {
791 kf = cv::KalmanFilter(stateSize, measSize, contrSize, CV_32F);
792
793 // Transition State Matrix A
794 // Note: set dT at each processing step!
795 // [ 1 0 dT 0 0 0 ]
796 // [ 0 1 0 dT 0 0 ]
797 // [ 0 0 1 0 0 0 ]
798 // [ 0 0 0 1 0 0 ]
799 // [ 0 0 0 0 1 0 ]
800 // [ 0 0 0 0 0 1 ]
801 cv::setIdentity(kf.transitionMatrix);
802
803 // Measure Matrix H
804 // [ 1 0 0 0 0 0 ]
805 // [ 0 1 0 0 0 0 ]
806 // [ 0 0 0 0 1 0 ]
807 // [ 0 0 0 0 0 1 ]
808 kf.measurementMatrix = cv::Mat::zeros(measSize, stateSize, CV_32F);
809 kf.measurementMatrix.at<float>(0) = 1.0f;
810 kf.measurementMatrix.at<float>(7) = 1.0f;
811 kf.measurementMatrix.at<float>(16) = 1.0f;
812 kf.measurementMatrix.at<float>(23) = 1.0f;
813
814 // Process Noise Covariance Matrix Q - result smoother with lower values (1e-2)
815 // [ Ex 0 0 0 0 0 ]
816 // [ 0 Ey 0 0 0 0 ]
817 // [ 0 0 Ev_x 0 0 0 ]
818 // [ 0 0 0 Ev_y 0 0 ]
819 // [ 0 0 0 0 Ew 0 ]
820 // [ 0 0 0 0 0 Eh ]
821 //cv::setIdentity(kf.processNoiseCov, cv::Scalar(1e-3));
822 kf.processNoiseCov.at<float>(0) = 1e-2;
823 kf.processNoiseCov.at<float>(7) = 1e-2;
824 kf.processNoiseCov.at<float>(14) = 1e-2;// 5.0f;
825 kf.processNoiseCov.at<float>(21) = 1e-2;// 5.0f;
826 kf.processNoiseCov.at<float>(28) = 5e-3;
827 kf.processNoiseCov.at<float>(35) = 5e-3;
828
829 // Measures Noise Covariance Matrix R - result smoother with higher values (1e-1)
830 cv::setIdentity(kf.measurementNoiseCov, cv::Scalar(1e-1));
831
832 //cv::setIdentity(kf.errorCovPost, cv::Scalar::all(1e-2));
833 // <<<< Kalman Filter
834
835 set_delta_time(0);
836 }
