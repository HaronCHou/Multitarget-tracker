101void get_lin_regress_params(
102 const CONT& in_data,
103 size_t start_pos,
104 size_t in_data_size,
105 T& kx, T& bx, T& ky, T& by)
106{
107 T m1(0.), m2(0.);
108 T m3_x(0.), m4_x(0.);
109 T m3_y(0.), m4_y(0.);
110
111 const T el_count = static_cast<T>(in_data_size - start_pos);
112 for (size_t i = start_pos; i < in_data_size; ++i)
113 {
114 m1 += i;
115 m2 += sqr(i);
116
117 m3_x += in_data[i].x;
118 m4_x += i * in_data[i].x;
119
120 m3_y += in_data[i].y;
121 m4_y += i * in_data[i].y;
122 }
123 T det_1 = 1 / (el_count * m2 - sqr(m1));
124
125 m1 *= -1;
126
127 kx = det_1 * (m1 * m3_x + el_count * m4_x);
128 bx = det_1 * (m2 * m3_x + m1 * m4_x);
129
130 ky = det_1 * (m1 * m3_y + el_count * m4_y);
131 by = det_1 * (m2 * m3_y + m1 * m4_y);
132}
