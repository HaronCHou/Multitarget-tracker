215 std::vector<cv::Rect> GetCrops(float maxCropRatio, cv::Size netSize, cv::Size imgSize) const
216 {
217 std::vector<cv::Rect> crops;
218
219 const float whRatio = static_cast<float>(netSize.width) / static_cast<float>(netSize.height);
220 int cropHeight = cvRound(maxCropRatio * netSize.height);
221 int cropWidth = cvRound(maxCropRatio * netSize.width);
222
223 if (imgSize.width / (float)imgSize.height > whRatio)
224 {
225 if (cropHeight >= imgSize.height)
226 cropHeight = imgSize.height;
227 cropWidth = cvRound(cropHeight * whRatio);
228 }
229 else
230 {
231 if (cropWidth >= imgSize.width)
232 cropWidth = imgSize.width;
233 cropHeight = cvRound(cropWidth / whRatio);
234 }
235
236 //std::cout << "Frame size " << imgSize << ", crop size = " << cv::Size(cropWidth, cropHeight) << ", ratio = " << maxCropRatio << std::endl;
237
238 const int stepX = 3 * cropWidth / 4;
239 const int stepY = 3 * cropHeight / 4;
240 for (int y = 0; y < imgSize.height; y += stepY)
241 {
242 bool needBreakY = false;
243 if (y + cropHeight >= imgSize.height)
244 {
245 y = imgSize.height - cropHeight;
246 needBreakY = true;
247 }
248 for (int x = 0; x < imgSize.width; x += stepX)
249 {
250 bool needBreakX = false;
251 if (x + cropWidth >= imgSize.width)
252 {
253 x = imgSize.width - cropWidth;
254 needBreakX = true;
255 }
256 crops.emplace_back(x, y, cropWidth, cropHeight);
257 if (needBreakX)
258 break;
259 }
260 if (needBreakY)
261 break;
262 }
263 return crops;
264 }
