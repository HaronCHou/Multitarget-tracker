535template <class T>
536symlist_iterator<T, T&> symlist<T>::insert(
537 symlist_iterator<T,T&> pos,
538 const T& ins)
539{
540 iterator prev = pos;
541 --prev;
542 symnode<T>* n = new symnode<T>(ins);
543 n->adj[0] = pos.act;
544 n->adj[1] = prev.act;
545
546 if (pos == prev)
547 {
548 pos = prev;
549 }
550
551 pos.prev() = n;
552 prev.next() = n;
553
554 return iterator(n, 0);
555}

558template <class T>
559void symlist<T>::splice(symlist_iterator<T, T&> pos,
560 symlist_iterator<T, T&> beg,
561 symlist_iterator<T, T&> end)
562{
563 if (beg != end)
564 {
565 iterator prev = beg;
566 --prev;
567 iterator last = end;
568 --last;
569
570 //
571 // The following seems to be rather senseless, but it is required
572 // since two iterator are equal, iff the point to the same element.
573 // This implies that they might have different directions. Suppose
574 // that prev == end is true and they have different directions,
575 // than prev.next() and end.prev() return the same element !! Thus
576 // the assignment prev = end corrects this, since the direction
577 // gets copied, too.
578 //
579 if (prev == end)
580 {
581 prev = end;
582 }
583
584 prev.next() = end.act;
585 end.prev() = prev.act;
586
587 prev = pos;
588 --prev;
589
590 if (pos == prev)
591 {
592 pos = prev;
593 }
594
595 if (last == beg)
596 {
597 last = beg;
598 }
599
600 prev.next() = beg.act;
601 beg.prev() = prev.act;
602 pos.prev() = last.act;
603 last.next() = pos.act;
604 }
605}

618template <class T>
619symlist_iterator<T,T&> symlist<T>::erase(symlist_iterator<T,T&> pos)
620{
621 assert (pos.act != link);
622 iterator prev = pos;
623 --prev;
624 iterator next = pos;
625 ++next;
626
627 if (next == prev)
628 {
629 next = prev;
630 }
631
632 next.prev() = prev.act;
633 prev.next() = next.act;
634
635 delete (pos.act);
636
637 return next;
638}

640template <class T>
641symlist_iterator<T,T&> symlist<T>::erase(symlist_iterator<T,T&> beg,
642 symlist_iterator<T,T&> end)
643{
644 iterator prev = beg;
645 --prev;
646 iterator it = beg;
647 symnode<T>* act;
648
649 while (it != end)
650 {
651 assert (it.act != link);
652 act = it.act;
653 ++it;
654 delete (act);
655 }
656
657 if (prev == end)
658 {
659 prev = end;
660 }
661
662 end.prev() = prev.act;
663 prev.next() = end.act;
664
665 return end;
666}

669template <class T>
670void symlist<T>::attach_sublist(symlist_iterator<T,T&> it,
671 symlist_iterator<T,T&> end)
672{
673 assert (empty());
674 iterator last = end;
675 --last;
676 _prev = it;
677 --_prev;
678 _next = end;
679
680 if (it == last)
681 {
682 it = last;
683 }
684
685 link->adj[0] = it.act;
686 it.prev() = link;
687 link->adj[1] = last.act;
688 last.next() = link;
689}

692template <class T>
693void symlist<T>::detach_sublist()
694{
695 if (_next != iterator())
696 {
697 iterator it(begin());
698 iterator e(end());
699
700 --e;
701
702 if (e == it)
703 {
704 e = it;
705 }
706
707 _prev.next() = it.act;
708 it.prev() = _prev.act;
709 _next.prev() = e.act;
710 e.next() = _next.act;
711 link->adj[0] = link->adj[1] = link;
712
713 _next = iterator();
714 _prev = iterator();
715 }
716}
