186 void calcuate_letterbox_message(const int m_InputH, const int m_InputW,
187 const int imageH, const int imageW,
188 float &sh,float &sw,
189 int &xOffset,int &yOffset)
190 {
191 float r = std::min(static_cast<float>(m_InputH) / static_cast<float>(imageH), static_cast<float>(m_InputW) / static_cast<float>(imageW));
192 int resizeH = (std::round(imageH*r));
193 int resizeW = (std::round(imageW*r));
194
195 sh = r;
196 sw = r;
197 if ((m_InputW - resizeW) % 2) resizeW--;
198 if ((m_InputH - resizeH) % 2) resizeH--;
199 assert((m_InputW - resizeW) % 2 == 0);
200 assert((m_InputH - resizeH) % 2 == 0);
201 xOffset = (m_InputW - resizeW) / 2;
202 yOffset = (m_InputH - resizeH) / 2;
203 }
204 BBox convert_bbox_res(const float& bx, const float& by, const float& bw, const float& bh,
205 const uint32_t& stride_h_, const uint32_t& stride_w_, const uint32_t& netW, const uint32_t& netH)
206 {
207 BBox b;
208 // Restore coordinates to network input resolution
209 float x = bx * stride_w_;
210 float y = by * stride_h_;
211
212 b.x1 = x - bw / 2;
213 b.x2 = x + bw / 2;
214
215 b.y1 = y - bh / 2;
216 b.y2 = y + bh / 2;
217
218 b.x1 = clamp(b.x1, 0.f, static_cast<float>(netW));
219 b.x2 = clamp(b.x2, 0.f, static_cast<float>(netW));
220 b.y1 = clamp(b.y1, 0.f, static_cast<float>(netH));
221 b.y2 = clamp(b.y2, 0.f, static_cast<float>(netH));
222
223 return b;
224 }

226 inline void cvt_box(const float sh,
227 const float sw,
228 const float xOffset,
229 const float yOffset,
230 BBox& bbox)
231 {
233 bbox.x1 -= xOffset;
234 bbox.x2 -= xOffset;
235 bbox.y1 -= yOffset;
236 bbox.y2 -= yOffset;
238 bbox.x1 /= sw;
239 bbox.x2 /= sw;
240 bbox.y1 /= sh;
241 bbox.y2 /= sh;
242 }

244 inline void add_bbox_proposal(const float bx, const float by, const float bw, const float bh,
245 const uint32_t stride_h_, const uint32_t stride_w_, const float scaleH, const float scaleW, const float xoffset_, const float yoffset, const int maxIndex, const float maxProb,
246 const uint32_t image_w, const uint32_t image_h,
247 std::vector<BBoxInfo>& binfo)
248 {
249 BBoxInfo bbi;
250 bbi.box = convert_bbox_res(bx, by, bw, bh, stride_h_, stride_w_, m_InputW, m_InputH);
251 if ((bbi.box.x1 > bbi.box.x2) || (bbi.box.y1 > bbi.box.y2))
252 {
253 return;
254 }
255 if ("yolov5" == m_NetworkType)
256 {
257 cvt_box(scaleH, scaleW, xoffset_, yoffset, bbi.box);
258 }
259 else
260 {
261 bbi.box.x1 = ((float)bbi.box.x1 / (float)m_InputW)*(float)image_w;
262 bbi.box.y1 = ((float)bbi.box.y1 / (float)m_InputH)*(float)image_h;
263 bbi.box.x2 = ((float)bbi.box.x2 / (float)m_InputW)*(float)image_w;
264 bbi.box.y2 = ((float)bbi.box.y2 / (float)m_InputH)*(float)image_h;
265 }
266
267 bbi.label = maxIndex;
268 bbi.prob = maxProb;
269 bbi.classId = getClassId(maxIndex);
270 binfo.push_back(bbi);
271 };
