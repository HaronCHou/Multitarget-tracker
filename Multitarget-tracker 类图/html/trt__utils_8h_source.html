81 void log(nvinfer1::ILogger::Severity severity, const char* msg) noexcept override
82 {
83 // suppress info-level messages
84 if (severity == Severity::kINFO) return;
85
86 switch (severity)
87 {
88 case Severity::kINTERNAL_ERROR: std::cerr << "INTERNAL_ERROR: " << msg << std::endl; break;
89 case Severity::kERROR: std::cerr << "ERROR: " << msg << std::endl; break;
90 case Severity::kWARNING: std::cerr << "WARNING: " << msg << std::endl; break;
91 case Severity::kINFO: std::cerr << "INFO: " << msg << std::endl; break;
92 case Severity::kVERBOSE: break;
93 // default: std::cerr <<"UNKNOW:"<< msg << std::endl;break;
