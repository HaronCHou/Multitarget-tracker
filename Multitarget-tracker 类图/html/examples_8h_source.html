12void DrawFilledRect(cv::Mat& frame, const cv::Rect& rect, cv::Scalar cl, int alpha)
13{
14 if (alpha)
15 {
16 const int alpha_1 = 255 - alpha;
17 const int nchans = frame.channels();
18 int color[3] = { cv::saturate_cast<int>(cl[0]), cv::saturate_cast<int>(cl[1]), cv::saturate_cast<int>(cl[2]) };
19 for (int y = rect.y; y < rect.y + rect.height; ++y)
20 {
21 uchar* ptr = frame.ptr(y) + nchans * rect.x;
22 for (int x = rect.x; x < rect.x + rect.width; ++x)
23 {
24 for (int i = 0; i < nchans; ++i)
25 {
26 ptr[i] = cv::saturate_cast<uchar>((alpha_1 * ptr[i] + alpha * color[i]) / 255);
27 }
28 ptr += nchans;
29 }
30 }
31 }
32 else
33 {
34 cv::rectangle(frame, rect, cl, cv::FILLED);
35 }
36}

187 brect.height = std::min(brect.height, frame.rows - 1);

193 brect.height = std::min(brect.height, frame.rows - 1);

201 0.7f, // Minimal ratio raw_trajectory_points / trajectory_lenght
202 cv::Size2f(0.1f, 8.0f))) // Min and max ratio: width / height

204 }
205 }
206 m_detector->CalcMotionMap(frame);
207 }
