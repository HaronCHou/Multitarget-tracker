155template <class Key, class Value, class Graph, class Alloc>
156typename ne_map<Key,Value,Graph,Alloc>::value_reference ne_map<Key,Value,Graph,Alloc>::operator[](Key t1)
157{
158 if(t1.id() >= (signed)data.size())
159 {
160 if (t1.id() >= (signed)data.capacity()) {
161 data.reserve((6 * t1.id()) / 5 + 1);
162 }
163
164 data.insert(data.end(), t1.id()+1-data.size(), Value());
165 }
166 return data.operator[](t1.id());
167}
